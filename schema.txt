Books(id as PRIMARY KEY,
      title,
      author_name,
      publisher_name,
      genre as FOREIGN KEY TO Genre.id,
      Stock (No. of books available or NO)
)

User(id as PRIMARY KEY,
      name,
      hashed_password,
      email_id,
      phone_number,
      address,
      favorite_book as FOREIGN KEY TO Books.id,
      favorite_genre as FOREIGN KEY TO Genre.id
)

Issue_history(user_id as FOREIGN KEY TO User.id,
              book_id as a FOREIGN KEY TO Books.id,
              issue_date,
              return_status
)

Genre(id as PRIMARY KEY,
      name
)
